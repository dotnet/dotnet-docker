ARG REPO=mcr.microsoft.com/dotnet/runtime

# Installer image
FROM $REPO:0.0.0-alpine3.XX-arm32v7 AS installer

ARG ACCESSTOKEN

# Retrieve ASP.NET Core
RUN aspnetcore_version=0.0.0 \
    && wget --header "Authorization: Bearer $ACCESSTOKEN" --header "x-ms-version: 2017-11-09" --output-document aspnetcore.tar.gz https://dotnetstage.blob.core.windows.net/aspnetcore/Runtime/$aspnetcore_version/aspnetcore-runtime-$aspnetcore_version-linux-musl-arm.tar.gz \
    && aspnetcore_sha512='{sha512_placeholder}' \
    && echo "$aspnetcore_sha512  aspnetcore.tar.gz" | sha512sum -c - \
    && mkdir --parents /dotnet \
    && tar --gzip --extract --no-same-owner --file aspnetcore.tar.gz --directory /dotnet ./shared/Microsoft.AspNetCore.App \
    && rm aspnetcore.tar.gz


# ASP.NET Core image
FROM $REPO:0.0.0-alpine3.XX-arm32v7

# ASP.NET Core version
ENV ASPNET_VERSION=0.0.0

COPY --from=installer ["/dotnet", "/usr/share/dotnet"]
